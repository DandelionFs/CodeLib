#include<iostream>
using namespace std;
int main()
{
	int N, n,z;
	int steps = 0;
	int outs = 0;
	cin >> N; //总人数
	cin >> z;//间隔数
	bool* p = new bool[N];
	for (int i = 0; i < N; i++) p[i] = true;
	while (outs < N - 1)
	{
		for (int i = 0; i < N; i++)
		{
			if (p[i] != false) steps++;
			if (steps % z == 0)
			{
				if (p[i] != false)
				{
					p[i] = false;
					outs += 1;
					steps = 0;
				}
			}
		}
	}
	for (int i = 0; i < N; i++) if (p[i] != false)cout << i + 1;
}
